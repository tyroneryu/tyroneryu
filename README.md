class TechStack:
    def __init__(self):
        self.operating_systems = ["CentOS", "KALI LINUX", "LINUX", "UBUNTU", "MACOS"]
        self.containerization = ["Docker", "Amazon AWS"]
        self.databases = ["MySQL", "MariaDB"]

        self.programming_languages = ["Python", "C", "C++", "HTML5", "JavaScript", "R"]
        self.frameworks_and_libraries = ["VUE.JS", "DJANGO", "REACT"]
        self.version_control = ["GIT"]
        self.ai_ml = ["ChatGPT"]
        self.markup_languages = ["Markdown"]

    def display_stack(self):
        print("Operating Systems:", ", ".join(self.operating_systems))
        print("Containerization:", ", ".join(self.containerization))
        print("Databases:", ", ".join(self.databases))
        print("Development Tools:", ", ".join(self.development_tools))
        print("Programming Languages:", ", ".join(self.programming_languages))
        print("Frameworks and Libraries:", ", ".join(self.frameworks_and_libraries))
        print("Version Control:", ", ".join(self.version_control))
        print("AI and Machine Learning:", ", ".join(self.ai_ml))
        print("Markup Languages:", ", ".join(self.markup_languages))

# Example usage
my_stack = TechStack()
my_stack.display_stack()
